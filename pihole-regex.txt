# ============================================================
# Pi-hole Regex Blocklist 
# ============================================================
# Как добавить в Pi-hole:
#   Вариант 1 (GUI): Group Management → Domains →
#                    вкладка "Regex filter" → Add domain
#   Вариант 2 (CLI): pihole --regex "pattern"
#   Вариант 3 (sqlite): напрямую в /etc/pihole/gravity.db
#
# Синтаксис: regex применяется к FQDN без точки в конце.
# Якорь ^ = начало строки, $ = конец строки, \. = буквальная точка.
#
# РАЗДЕЛ A — ТОЧНЫЕ (Precise)
# Минимальный риск ложных срабатываний.
# Матчат конкретные субдомены известных рекламных платформ.
# ============================================================

# --- Yandex: только трекинговые субдомены ---
(^|\.)mc\.yandex\.ru$
(^|\.)metrika\.yandex\.ru$

# --- Yandex AdFox: ads.adfox.ru и весь домен adfox.ru ---
(^|\.)adfox\.ru$

# --- skcrtxr.com: все три рекламных субдомена ---
^(grid|hb-bidder|cdn)\.skcrtxr\.com$

# --- hybrid.ai: все рекламные субдомены ---
^(bobid-ip|fpf|ssp)\.hybrid\.ai$

# --- buzzoola.com: exchange и tube ---
^(exchange|tube)\.buzzoola\.com$

# --- adriver.ru: pb и content ---
^(pb|content)\.adriver\.ru$

# --- sape.ru: только RTB/CDN субдомены (не весь sape.ru) ---
^(ssp-rtb|cdn-rtb)\.sape\.ru$

# --- rutarget.ru: только bidder субдомен ---
^adfox-hb-bidder\.rutarget\.ru$

# --- Google Analytics: оба домена ---
^(www\.)?google-analytics\.com$
^analytics\.google\.com$

# --- Google Tag Manager ---
^www\.googletagmanager\.com$

# --- Weborama: вся цепочка субдоменов ---
(^|\.)weborama\.(fr|com)$

# --- Acint: поведенческий таргетинг ---
(^|\.)acint\.net$

# --- Smi2: exad субдомен ---
^exad\.smi2\.ru$

# --- OTM-R: Yandex HB партнёр ---
^yhb\.p\.otm-r\.com$

# --- MediaToday RTB ---
^mediatoday\.ru$

# ============================================================
# РАЗДЕЛ B — ЭВРИСТИЧЕСКИЕ (Heuristic)
# Умеренный риск ложных срабатываний.
# Ловят новые/неизвестные домены по паттернам имён.
# Рекомендуется тестировать на своей сети перед включением.
# ============================================================

# --- .bid TLD — почти исключительно используется биддинговыми платформами ---
# Риск: низкий (.bid не популярен для легитимных сайтов)
^[a-z0-9-]+\.bid$

# --- Субдомены вида hb-bidder.* (header bidding bidder) ---
# Риск: низкий (паттерн специфичен для HB инфраструктуры)
^hb-bidder\.[a-z0-9-]+\.[a-z]{2,}$

# --- Субдомены вида ssp-rtb.* и dsp-rtb.* ---
# Риск: низкий
^(ssp-rtb|dsp-rtb|cdn-rtb)\.[a-z0-9-]+\.[a-z]{2,}$

# --- Yandex Header Bidding партнёры (yhb.* паттерн) ---
# Риск: низкий (yhb. = Yandex Header Bidding, нестандартный префикс)
^yhb\.[a-z0-9.-]+\.[a-z]{2,}$

# --- Субдомены вида adfox-*.* (Yandex AdFox биддеры) ---
# Риск: низкий
^adfox-[a-z0-9-]+\.[a-z0-9-]+\.[a-z]{2,}$

# ============================================================
# РАЗДЕЛ C — АГРЕССИВНЫЕ (Aggressive)
# Высокий риск ложных срабатываний.
# Включать только если готовы к возможным проблемам с сайтами.
# ============================================================

# --- Любой домен с субдоменом "adx." ---
# ВНИМАНИЕ: может сломать некоторые легитимные сервисы
# ^adx\.[a-z0-9-]+\.[a-z]{2,}$

# --- Любой домен с субдоменом "ssp." ---
# ВНИМАНИЕ: ssp используется не только в рекламе
# ^ssp\.[a-z0-9-]+\.[a-z]{2,}$

# --- Любой домен с субдоменом "hb." (header bidding) ---
# ВНИМАНИЕ: hb. могут использоваться в других контекстах
# ^hb\.[a-z0-9-]+\.[a-z]{2,}$

# --- LiveInternet счётчик (yadro CDN) ---
# ВНИМАНИЕ: yadro.ru используется для других сервисов
# (^|\.)yadro\.ru$
